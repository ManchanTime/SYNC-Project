MODULE user_command_0100 INPUT.

  CASE gv_okcode.
    WHEN 'DISP'.
      PERFORM set_main_data.
    WHEN 'EXE'.
      PERFORM execute_mrp.
    WHEN 'TAB1' OR 'TAB2' OR 'TAB3'.
      gc_tab-activetab = gv_okcode.
    WHEN 'BOMS'.
      CALL TRANSACTION 'ZC102PPR0010'.
    WHEN 'POPUP'.
      CALL SCREEN 110 STARTING AT 10 15.
      PERFORM set_value_initialize.
  ENDCASE.

ENDMODULE.
