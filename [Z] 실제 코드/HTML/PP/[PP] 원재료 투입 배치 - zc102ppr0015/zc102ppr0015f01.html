FORM create_docu .
  CLEAR : gs_prodc, gv_tabix.

  LOOP AT gt_prodc INTO gs_prodc.

      call function 'ZFM_CL102_FI_08'
        EXPORTING
          iv_pdono = gs_prodc-pdono

  ENDLOOP.
