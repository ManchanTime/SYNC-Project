MODULE user_command_0100 INPUT.

  CASE gv_okcode.
    WHEN 'SHIP'.
      PERFORM trans_shipment.
    WHEN 'EMERG'.
      PERFORM trans_emergen.
    WHEN 'SERC'.
      PERFORM set_search_condition.
      PERFORM display_del_header.
*-- From 판매오더(By 소연)
    WHEN 'ENTER'.
      IF gv_from = abap_true.
        PERFORM set_search_condition.
        PERFORM display_del_header.
      ENDIF.
  ENDCASE.

ENDMODULE.
