  LEAVE TO SCREEN 0.

*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0100 INPUT.

  CASE gv_okcode.
    WHEN 'SAVE'.
      PERFORM save_data.
    WHEN 'EXCE'.
      PERFORM excel_job.
    WHEN 'PRIN'.
      CALL SCREEN 120 STARTING AT 15 10.
  ENDCASE.

ENDMODULE.
