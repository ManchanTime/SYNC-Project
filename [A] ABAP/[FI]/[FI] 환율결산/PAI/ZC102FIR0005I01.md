*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0100 INPUT.
  CASE gv_okcode.
    WHEN 'FIND'.
      CALL TRANSACTION 'ZC102FI0010'.
    WHEN 'LOOK'.
      CALL TRANSACTION 'ZC102FIR0003'.
    WHEN 'SRCH'.
      CLEAR : gt_open, gt_clear,
              gt_line, gt_clear_line.
      PERFORM get_open_data.
      PERFORM filter_data.
      PERFORM set_icon.
      PERFORM refresh_table.
  ENDCASE.
ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0110  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0110 INPUT.
  CASE gv_okcode.
    WHEN 'EXCT'.
      PERFORM clear.
      gv_check = abap_true.
      PERFORM calculate_amount.
    WHEN 'CRTE'.
      IF gv_ex_amount IS NOT INITIAL.
        PERFORM create_docu.
        LEAVE TO SCREEN 0.
      ELSE.
        MESSAGE s000 WITH TEXT-s03 DISPLAY LIKE 'E'.
      ENDIF.
  ENDCASE.
ENDMODULE.
