MODULE user_command_0120 INPUT.
  CASE gv_okcode.
    WHEN 'SAVE'.
      PERFORM check_value.
      IF gv_create_lock = abap_false.
        PERFORM create_asset.
      ENDIF.

    WHEN 'ATCH'.
      PERFORM set_anln1 CHANGING gv_num.
      PERFORM call_gos USING gv_num '1000' 'C'.
      PERFORM check_upload.
  ENDCASE.
ENDMODULE.
